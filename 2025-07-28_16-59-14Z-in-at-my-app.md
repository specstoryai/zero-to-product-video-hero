import React from 'react';
import {AbsoluteFill, Video, staticFile, useCurrentFrame, interpolate} from 'remotion';
import {colors} from '../styles/design-tokens';

interface VideoIntroProps {
  src: string;
  startFrom?: number;
  endAt?: number;
}

export const VideoIntro: React.FC<VideoIntroProps> = ({ 
  src, 
  startFrom = 0,
  endAt
}) => {
  const frame = useCurrentFrame();
  
  // Fade in/out effect
  const opacity = interpolate(
    frame,
    [0, 20, (endAt || 120) - 20, endAt || 120],
    [0, 1, 1, 0],
    {
      extrapolateLeft: 'clamp',
      extrapolateRight: 'clamp'
    }
  );
  
  return (
    <AbsoluteFill style={{backgroundColor: colors.background.dark}}>
      <Video
        src={staticFile(src)}
        startFrom={startFrom}
        endAt={endAt}
        style={{
          width: '100%',
          height: '100%',
          objectFit: 'cover',
          opacity
        }}
      />
      
      {/* Optional overlay gradient for better transition */}
      <div
        style={{
          position: 'absolute',
          inset: 0,
          background: `linear-gradient(to bottom, transparent 70%, ${colors.background.dark} 100%)`,
          opacity: interpolate(
            frame,
            [(endAt || 120) - 30, endAt || 120],
            [0, 1],
            {
              extrapolateLeft: 'clamp',
              extrapolateRight: 'clamp'
            }
          )
        }}
      />
    </AbsoluteFill>
  );
};
